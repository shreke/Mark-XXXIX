COMMON QUERIES MAIA MUST KNOW:

-- Efectividad promedio hoy (UMILLA)
SELECT AVG(porcentaje_efectividad) FROM viajes_umilla WHERE fecha = CURRENT_DATE AND porcentaje_efectividad IS NOT NULL;

-- Choferes con efectividad baja (< 85%) en los últimos 7 días
SELECT nombres_chofer, AVG(porcentaje_efectividad) as efect_prom
FROM viajes_umilla
WHERE fecha >= CURRENT_DATE - INTERVAL '7 days' AND porcentaje_efectividad IS NOT NULL
GROUP BY nombres_chofer
HAVING AVG(porcentaje_efectividad) < 85
ORDER BY efect_prom ASC;

-- Viajes de hoy por estado
SELECT estado, COUNT(*) FROM viajes_umilla WHERE fecha = CURRENT_DATE GROUP BY estado;

-- Citaciones de hoy con detalle de chofer
SELECT ch.nombres, cc.estado, cm.hora_cita
FROM citaciones_choferes cc
JOIN citaciones_multiples cm ON cc.id_citacion = cm.id
JOIN choferes ch ON cc.id_chofer = ch.id
WHERE cm.fecha = CURRENT_DATE
ORDER BY cm.hora_cita;

-- Liquidaciones pendientes
SELECT COUNT(*), SUM(total_neto) FROM liquidaciones WHERE estado = 'PENDIENTE';

-- Gastos del mes actual
SELECT SUM(monto), COUNT(*) FROM gastos
WHERE DATE_TRUNC('month', fecha) = DATE_TRUNC('month', CURRENT_DATE);

-- Tickets abiertos por prioridad
SELECT prioridad, COUNT(*) FROM tickets WHERE estado != 'FINALIZADO' GROUP BY prioridad;

-- Vencimientos próximos de documentos (choferes, 30 días)
SELECT nombres, fecha_vto_registro, fecha_vto_rnr, vencimiento_linti
FROM choferes
WHERE estado = 'ACTIVO'
AND (fecha_vto_registro <= CURRENT_DATE + INTERVAL '30 days'
  OR fecha_vto_rnr <= CURRENT_DATE + INTERVAL '30 days'
  OR vencimiento_linti <= CURRENT_DATE + INTERVAL '30 days');
